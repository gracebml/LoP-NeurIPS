\section{Mục tiêu}

Tài liệu này thay bản chứng minh cũ bằng một phiên bản chặt hơn. Điểm chính là
phải tách riêng hai tầng lập luận:
\begin{enumerate}[label=(\roman*)]
    \item phần lõi thật sự suy ra từ cấu trúc Fisher của cặp neuron gần-clone;
    \item phần khuếch đại NGD, vốn cần thêm một giả thiết cấu trúc về hướng phá
    đối xứng trong \emph{toàn bộ} ma trận Fisher, chứ không chỉ trong block
    $2\times 2$.
\end{enumerate}

Bản cũ đã nhìn đúng trực giác, nhưng có vài bước nhảy logic chưa được biện minh.
Tài liệu này viết lại phần chứng minh ở dạng có thể kiểm tra từng bước và liệt
kê riêng các lỗi lập luận ở cuối.

\section{Thiết lập}

Xét mạng ReLU một tầng ẩn
\[
f(x) = \sum_{k=1}^m v_k \max(0, w_k^\top x),
\]
trong đó $w_k \in \R^d$ và $v_k \in \R$.

Ta xét hai neuron $i,j$ thỏa
\[
w_j = w_i - \delta \norm{w_i}\hat e,
\qquad
v_i = v_j,
\qquad
\delta > 0 \text{ nhỏ},
\]
với $\hat e$ là vector đơn vị. Đặt
\[
\ell'(x) := \frac{\partial \cL}{\partial f}(x),
\qquad
u_- := \frac{e_i - e_j}{\sqrt 2} \in \R^m.
\]

Với K-FAC ở tầng này, ta dùng hai ma trận thống kê
\[
A = \E[xx^\top],
\qquad
G(\delta) = \E[g g^\top],
\]
trong đó
\[
g_k = \ell'(x)\, v_k\, \mathbf 1(w_k^\top x > 0).
\]

Đặt hai miền bất đồng cổng ReLU
\[
D_{ij} = \{x : w_i^\top x > 0 \ge w_j^\top x\},
\qquad
D_{ji} = \{x : w_j^\top x > 0 \ge w_i^\top x\},
\]
và
\[
\mathrm{strip} = D_{ij} \cup D_{ji},
\qquad
P_{\mathrm{strip}}(\delta) = \Pr(x \in \mathrm{strip}),
\]
\[
B_\ell^{\mathrm{strip}}(\delta)
=
\E\!\left[\ell'(x)^2 \middle| x \in \mathrm{strip}\right].
\]

\begin{assumption}[Hiệp phương sai đầu vào xác định dương]
\[
A = \E[xx^\top] \succ 0.
\]
\end{assumption}

\begin{assumption}[Đạo hàm loss bị chặn]
Tồn tại $B_y < \infty$ sao cho
\[
\abs{\ell'(x)} \le B_y
\qquad \text{gần như chắc chắn.}
\]
\end{assumption}

\begin{assumption}[Scale tuyến tính của strip và tín hiệu không suy biến trên strip]
\label{ass:strip_scaling}
Tồn tại các hằng số $c_-, c_+, b_-, b_+ > 0$ sao cho với mọi $\delta$ đủ nhỏ,
\[
c_-\,\delta \le P_{\mathrm{strip}}(\delta) \le c_+\,\delta,
\qquad
b_- \le B_\ell^{\mathrm{strip}}(\delta) \le b_+.
\]
\end{assumption}

\begin{remark}
Giả thiết~\ref{ass:strip_scaling} mạnh hơn câu ``mật độ bị chặn gần siêu phẳng''.
Nó được đưa vào trực tiếp vì đó mới là điều thật sự cần để suy ra
$P_{\mathrm{strip}} = \Theta(\delta)$ và $u_-^\top G u_- = \Theta(\delta)$.
Bản cũ đã dùng đúng kết luận này nhưng chưa có giả thiết đủ mạnh để suy ra.
\end{remark}

\begin{assumption}[Hướng phá đối xứng là eigenvector của Fisher]
\label{ass:eigenvector}
Với mọi $\delta$ đủ nhỏ, tồn tại $\lambda_-(\delta) \ge 0$ sao cho
\[
G(\delta) u_- = \lambda_-(\delta) u_-.
\]
\end{assumption}

\begin{remark}
Giả thiết~\ref{ass:eigenvector} chính là phần cấu trúc mà bước NGD cần. Nếu chỉ
biết block $2\times 2$ ứng với $(i,j)$ có eigenvalue nhỏ mà không kiểm soát sự
coupling của hai neuron này với các neuron còn lại, thì không thể suy ra công
thức nghịch đảo của toàn ma trận $(G+\gamma I)^{-1}$ theo hướng $u_-$.
\end{remark}

\section{Phát biểu đã chỉnh}

\begin{theorem}[Phiên bản chặt cho symmetry breaking của cloned units]
\label{thm:clone_revised}
Dưới các Giả thiết trên, ta có:
\begin{enumerate}[label=(\roman*)]
    \item \textbf{Độ cong Fisher theo hướng phản đối xứng có công thức chính xác.}
    Nếu đặt
    \[
    \sigma_{\mathrm{clone}}^{(ij)}(\delta)
    :=
    G_{ii}(\delta) + G_{jj}(\delta) - 2G_{ij}(\delta),
    \]
    thì
    \[
    \sigma_{\mathrm{clone}}^{(ij)}(\delta)
    =
    v_i^2 B_\ell^{\mathrm{strip}}(\delta) P_{\mathrm{strip}}(\delta),
    \]
    và do đó
    \[
    \lambda_-(\delta) = u_-^\top G(\delta) u_-
    = \frac{1}{2} v_i^2 B_\ell^{\mathrm{strip}}(\delta) P_{\mathrm{strip}}(\delta)
    = \Theta(\delta).
    \]

    \item \textbf{Công thức NGD đúng chính xác dưới giả thiết eigenvector.}
    Với damping $\gamma > 0$,
    \[
    \Delta W^{\mathrm{NGD}}
    =
    -(G(\delta)+\gamma I_m)^{-1}\,\nabla_W \cL\,(A+\gamma I_d)^{-1}.
    \]
    Khi đó ta có đẳng thức chính xác
    \[
    \Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}
    =
    \frac{1}{\gamma + \lambda_-(\delta)}
    \bigl(\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}\bigr)
    (A+\gamma I_d)^{-1},
    \]
    trong đó ta bỏ cùng learning rate ở hai vế.

    \item \textbf{Hệ quả khuếch đại.}
    Ta có cận dưới
    \[
    \frac{\norm{\Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}}}
         {\norm{\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}}}
    \ge
    \frac{1}{(\gamma + \lambda_-(\delta))\,\norm{A+\gamma I}_{\op}}.
    \]
    Vì $\lambda_-(\delta)=\Theta(\delta) \to 0$, nên
    \[
    \liminf_{\delta \to 0}
    \frac{\norm{\Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}}}
         {\norm{\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}}}
    \ge
    \frac{1}{\gamma\,\norm{A+\gamma I}_{\op}}
    =
    \frac{1}{\gamma(\lambda_{\max}(A)+\gamma)}.
    \]
\end{enumerate}
\end{theorem}

\section{Chứng minh}

\subsection*{Bước 1: Viết rõ các phần tử của $G$}

Từ định nghĩa $g_k = \ell'(x) v_k \mathbf 1(w_k^\top x > 0)$, ta có
\begin{align}
G_{ii}
&=
\E[g_i^2]
=
v_i^2 \E\!\left[\ell'(x)^2 \mathbf 1(w_i^\top x > 0)\right], \\
G_{jj}
&=
\E[g_j^2]
=
v_i^2 \E\!\left[\ell'(x)^2 \mathbf 1(w_j^\top x > 0)\right], \\
G_{ij}
&=
\E[g_i g_j]
=
v_i^2 \E\!\left[\ell'(x)^2 \mathbf 1(w_i^\top x > 0)\mathbf 1(w_j^\top x > 0)\right].
\end{align}

Ở đây ta đã dùng $v_i=v_j$.

\subsection*{Bước 2: Công thức chính xác cho độ cong clone}

Ta có
\begin{align}
G_{ii} - G_{ij}
&=
v_i^2 \E\!\left[\ell'(x)^2 \mathbf 1(w_i^\top x > 0)\bigl(1-\mathbf 1(w_j^\top x > 0)\bigr)\right] \\
&=
v_i^2 \E\!\left[\ell'(x)^2 \mathbf 1_{D_{ij}}(x)\right].
\end{align}
Tương tự,
\[
G_{jj} - G_{ij}
=
v_i^2 \E\!\left[\ell'(x)^2 \mathbf 1_{D_{ji}}(x)\right].
\]
Cộng hai đẳng thức,
\[
\sigma_{\mathrm{clone}}^{(ij)}(\delta)
=
G_{ii}+G_{jj}-2G_{ij}
=
v_i^2 \E\!\left[\ell'(x)^2 \mathbf 1_{\mathrm{strip}}(x)\right].
\]
Theo định nghĩa của kỳ vọng có điều kiện,
\[
\sigma_{\mathrm{clone}}^{(ij)}(\delta)
=
v_i^2 B_\ell^{\mathrm{strip}}(\delta) P_{\mathrm{strip}}(\delta).
\]

Mặt khác, vì $u_-=(e_i-e_j)/\sqrt 2$,
\[
u_-^\top G(\delta) u_-
=
\frac{1}{2}\bigl(G_{ii}+G_{jj}-2G_{ij}\bigr)
=
\frac{1}{2}\sigma_{\mathrm{clone}}^{(ij)}(\delta).
\]
Do đó
\[
\lambda_-(\delta)
=
u_-^\top G(\delta) u_-
=
\frac{1}{2} v_i^2 B_\ell^{\mathrm{strip}}(\delta) P_{\mathrm{strip}}(\delta).
\]
Áp dụng Giả thiết~\ref{ass:strip_scaling}, ta suy ra ngay
\[
\lambda_-(\delta) = \Theta(\delta).
\]

\subsection*{Bước 3: Công thức chính xác cho hiệu cập nhật NGD}

Đặt $r := e_i-e_j = \sqrt 2\,u_-$. Khi đó hiệu hai hàng cập nhật thỏa
\[
\Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}
=
-r^\top (G(\delta)+\gamma I_m)^{-1} \nabla_W \cL\,(A+\gamma I_d)^{-1}.
\]

Do $G(\delta)$ đối xứng và theo Giả thiết~\ref{ass:eigenvector} ta có
$G(\delta)u_- = \lambda_-(\delta)u_-$, suy ra
\[
(G(\delta)+\gamma I_m)^{-1}u_-
=
\frac{1}{\gamma+\lambda_-(\delta)}u_-.
\]
Nhân với $\sqrt 2$ cho $r$, ta được
\[
r^\top (G(\delta)+\gamma I_m)^{-1}
=
\frac{1}{\gamma+\lambda_-(\delta)}r^\top.
\]
Thế vào công thức trên,
\[
\Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}
=
-\frac{1}{\gamma+\lambda_-(\delta)}
r^\top \nabla_W \cL\,(A+\gamma I_d)^{-1}.
\]
Với SGD,
\[
\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}
=
-r^\top \nabla_W \cL,
\]
(nếu bỏ cùng learning rate ở cả hai vế). Vậy
\[
\Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}
=
\frac{1}{\gamma+\lambda_-(\delta)}
\bigl(\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}\bigr)
(A+\gamma I_d)^{-1}.
\]
Đây là điều phải chứng minh trong phần (ii).

\subsection*{Bước 4: Cận dưới cho tỉ số khuếch đại}

Với mọi row vector $z$ và ma trận đối xứng xác định dương $M$, ta có
\[
\norm{zM^{-1}} \ge \frac{\norm{z}}{\norm{M}_{\op}}.
\]
Áp dụng với $M=A+\gamma I_d$ và
\[
z = \frac{1}{\gamma+\lambda_-(\delta)}
\bigl(\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}\bigr),
\]
ta được
\[
\norm{\Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}}
\ge
\frac{1}{(\gamma+\lambda_-(\delta))\,\norm{A+\gamma I}_{\op}}
\norm{\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}}.
\]
Chia hai vế cho chuẩn ở mẫu, ta thu được phần (iii).

Cuối cùng, vì $\lambda_-(\delta)=\Theta(\delta) \to 0$ khi $\delta \to 0$,
\[
\liminf_{\delta \to 0}
\frac{\norm{\Delta w_i^{\mathrm{NGD}} - \Delta w_j^{\mathrm{NGD}}}}
     {\norm{\Delta w_i^{\mathrm{SGD}} - \Delta w_j^{\mathrm{SGD}}}}
\ge
\frac{1}{\gamma\,\norm{A+\gamma I}_{\op}}
=
\frac{1}{\gamma(\lambda_{\max}(A)+\gamma)}.
\]
Chứng minh hoàn tất. \qed

\section{Các lỗi lập luận trong bản cũ}

Dưới đây là các lỗi hoặc bước thiếu giả thiết trong bản trước.

\begin{enumerate}[label=\arabic*.]
    \item \textbf{Suy ra $P_{\mathrm{strip}} \le 2\rho_{\max}\delta\norm{w_i}$ từ giả thiết mật độ một chiều là chưa đúng.}
    Từ
    \[
    w_j = w_i - \delta\norm{w_i}\hat e
    \]
    ta chỉ suy ra sự bất đồng cổng nằm trong tập
    \[
    \abs{w_i^\top x} \le \delta\norm{w_i}\abs{\hat e^\top x}.
    \]
    Bề dày của strip vì thế phụ thuộc vào biến ngẫu nhiên $\abs{\hat e^\top x}$; nó
    không phải một dải có bề dày hằng bằng $\delta\norm{w_i}$. Muốn suy ra
    $P_{\mathrm{strip}} = O(\delta)$ cần thêm kiểm soát tail hoặc mật độ có điều kiện,
    chứ giả thiết một chiều trong bản cũ là chưa đủ.

    \item \textbf{Cận dưới cho $P_{\mathrm{strip}}$ cũng chưa suy ra được.}
    Từ việc có khối lượng gần siêu phẳng $\{w_i^\top x = 0\}$ không thể kết luận ngay
    rằng lượng khối đó thực sự bị đẩy qua hai phía của biên khi thay $w_i$ bằng $w_j$.
    Cần thêm thông tin rằng $\hat e^\top x$ không suy biến trên biên kích hoạt.

    \item \textbf{Kết luận $\sigma_{\mathrm{clone}}^{(ij)} = \Theta(\delta)$ cần một cận dưới đều cho $B_\ell^{\mathrm{strip}}(\delta)$.}
    Việc viết ``$B_\ell^{\mathrm{strip}} > 0$'' là chưa đủ, vì đại lượng này có thể vẫn
    tiến về $0$ khi $\delta \to 0$. Muốn có $\Theta(\delta)$ thật sự, cần giả thiết
    $b_- \le B_\ell^{\mathrm{strip}}(\delta)$ với $b_->0$ độc lập với $\delta$.

    \item \textbf{Bước từ Rayleigh quotient sang eigenvalue nhỏ nhất của block $2\times 2$ chỉ là heuristic.}
    Từ
    \[
    u_-^\top G u_- = \frac{1}{2}(G_{ii}+G_{jj}-2G_{ij})
    \]
    chỉ suy ra một Rayleigh quotient trên hướng $u_-$. Để đồng nhất nó với
    eigenvalue nhỏ nhất của block $2\times 2$, cần thêm đối xứng như
    $G_{ii}=G_{jj}$ hoặc phải chứng minh trực tiếp rằng $u_-$ là eigenvector của block.

    \item \textbf{Bước NGD trong bản cũ đã dùng thêm giả thiết mà chưa phát biểu.}
    Các câu như ``$u_-$ align chủ yếu với eigenvector nhỏ nhất'' và ``gradient khác biệt
    tập trung chủ yếu ở hướng $u_-$'' là giả thiết bổ sung, không phải hệ quả từ các giả
    thiết ban đầu. Nếu không phát biểu chúng rõ ràng thì bất đẳng thức khuếch đại chưa
    được chứng minh.

    \item \textbf{Không thể suy ra công thức cho $(G+\gamma I)^{-1}$ của toàn ma trận chỉ từ block $2\times 2$.}
    Nghịch đảo của ma trận đầy đủ còn phụ thuộc vào coupling của hai neuron $(i,j)$ với
    tất cả neuron khác. Vì vậy, từ một block nhỏ có eigenvalue cỡ $\Theta(\delta)$ chưa
    đủ để kết luận công thức cập nhật NGD cho toàn tầng. Đây là lý do bản chỉnh phải đưa
    vào Giả thiết~\ref{ass:eigenvector}.
\end{enumerate}

\section{Kết luận}

Điểm chắc nhất có thể rút ra là:
\begin{enumerate}[label=(\roman*)]
    \item độ cong Fisher theo hướng phá đối xứng đúng là do vùng bất đồng cổng quyết định;
    \item nếu strip có khối lượng tuyến tính theo $\delta$, thì độ cong đó là $\Theta(\delta)$;
    \item để nâng kết quả này thành công thức khuếch đại NGD cho toàn tầng, bắt buộc phải
    thêm một giả thiết cấu trúc kiểu ``hướng $u_-$ là mode riêng của Fisher'' hoặc một giả
    thiết tương đương đủ mạnh để kiểm soát coupling với các neuron còn lại.
\end{enumerate}

Nếu muốn giữ đúng phát biểu gốc trong \texttt{paper\_merged.tex}, thì nên phát biểu rõ
các giả thiết bổ sung này. Nếu không, lựa chọn an toàn hơn là chỉ giữ phần lõi:
$u_-^\top G u_- = \Theta(\delta)$ và mô tả bước NGD như một hệ quả dưới giả thiết cấu trúc
bổ sung.
